511  template<typename T, int C = 4, int R = 4>
512  inline std::enable_if_t<
513  (C >= 3 && R >= 3),
514  mat<decltype(tue::math::sin(std::declval<T>())), C, R>>
515  rotation_mat(
516  const T& axis_x, const T& axis_y, const T& axis_z,
517  const T& angle) noexcept
518  {
519  T s, c;
520  tue::math::sincos(angle, s, c);
521  const auto omc = T(1) - c;
522 
523  const auto xx = axis_x * axis_x;
524  const auto xy = axis_x * axis_y;
525  const auto xz = axis_x * axis_z;
526  const auto xs = axis_x * s;
527  const auto yy = axis_y * axis_y;
528  const auto yz = axis_y * axis_z;
529  const auto ys = axis_y * s;
530  const auto zz = axis_z * axis_z;
531  const auto zs = axis_z * s;
532  const auto xxomc = xx * omc;
533  const auto xyomc = xy * omc;
534  const auto xzomc = xz * omc;
535  const auto yyomc = yy * omc;
536  const auto yzomc = yz * omc;
537  const auto zzomc = zz * omc;
538 
539  return tue::detail_::mat_utils<T, C, R>::create(
540  xxomc + c, xyomc - zs, xzomc + ys, 0,
541  xyomc + zs, yyomc + c, yzomc - xs, 0,
542  xzomc - ys, yzomc + xs, zzomc + c, 0,
543  0, 0, 0, 1);
