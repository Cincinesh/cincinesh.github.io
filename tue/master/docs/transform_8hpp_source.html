90  axis_angle(const T& x, const T& y, const T& z) noexcept
91  {
92  const auto angle = tue::math::sqrt(x*x + y*y + z*z);
93  const auto nzmask = tue::math::not_equal(angle, T(0));
94 
95  const auto axis_x = tue::math::select(

98  nzmask, y / angle, T(0));

100  nzmask, z / angle, T(1));
101 
102  return { axis_x, axis_y, axis_z, angle };

480  template<typename T, int C = 4, int R = 4>
481  inline std::enable_if_t<(C >= 3 && R >= 3),
482  mat<decltype(tue::math::sin(std::declval<T>())), C, R>>
483  rotation_mat(
484  const T& axis_x, const T& axis_y, const T& axis_z,
485  const T& angle) noexcept
486  {
487  T s, c;
488  tue::math::sincos(angle, s, c);
489  const auto omc = T(1) - c;
490 
491  const auto xx = axis_x * axis_x;
492  const auto xy = axis_x * axis_y;
493  const auto xz = axis_x * axis_z;
494  const auto xs = axis_x * s;
495  const auto yy = axis_y * axis_y;
496  const auto yz = axis_y * axis_z;
497  const auto ys = axis_y * s;
498  const auto zz = axis_z * axis_z;
499  const auto zs = axis_z * s;
500  const auto xxomc = xx * omc;
501  const auto xyomc = xy * omc;
502  const auto xzomc = xz * omc;
503  const auto yyomc = yy * omc;
504  const auto yzomc = yz * omc;
505  const auto zzomc = zz * omc;
506 
507  return tue::detail_::mat_utils<T, C, R>::create(
508  xxomc + c, xyomc - zs, xzomc + ys, 0,
509  xyomc + zs, yyomc + c, yzomc - xs, 0,
510  xzomc - ys, yzomc + xs, zzomc + c, 0,
511  0, 0, 0, 1);
512  }
