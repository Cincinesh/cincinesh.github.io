510  template<typename T, int C = 4, int R = 4>
511  inline std::enable_if_t<(C >= 3 && R >= 3),
512  mat<decltype(tue::math::sin(std::declval<T>())), C, R>>
513  rotation_mat(
514  const T& axis_x, const T& axis_y, const T& axis_z,
515  const T& angle) noexcept
516  {
517  T s, c;
518  tue::math::sincos(angle, s, c);
519  const auto omc = T(1) - c;
520

522  const auto xy = axis_x * axis_y;
523  const auto xz = axis_x * axis_z;
524  const auto xs = axis_x * s;

526  const auto yz = axis_y * axis_z;
527  const auto ys = axis_y * s;
528  const auto zz = axis_z * axis_z;
529  const auto zs = axis_z * s;
530  const auto xxomc = xx * omc;
531  const auto xyomc = xy * omc;
532  const auto xzomc = xz * omc;
533  const auto yyomc = yy * omc;
534  const auto yzomc = yz * omc;
535  const auto zzomc = zz * omc;

537  return tue::detail_::mat_utils<T, C, R>::create(
538  xxomc + c, xyomc - zs, xzomc + ys, 0,
539  xyomc + zs, yyomc + c, yzomc - xs, 0,
540  xzomc - ys, yzomc + xs, zzomc + c, 0,
