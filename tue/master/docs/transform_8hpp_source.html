89  template<typename T>
90  inline vec4<decltype(tue::math::sqrt(std::declval<T>()))>
91  axis_angle(const T& x, const T& y, const T& z) noexcept
92  {
93  using U = decltype(tue::math::sqrt(x));
94  const auto u0 = U(0);
95  const auto u1 = U(1);
96 
97  const auto angle = tue::math::sqrt(x*x + y*y + z*z);
98  const auto nzmask = tue::math::not_equal(angle, u0);
99 
100  const auto axis_x = tue::math::select(
101  nzmask, tue::nocopy_cast<U>(x) / angle, u0);
102  const auto axis_y = tue::math::select(
103  nzmask, tue::nocopy_cast<U>(y) / angle, u0);
104  const auto axis_z = tue::math::select(
105  nzmask, tue::nocopy_cast<U>(z) / angle, u1);
106 
107  return { axis_x, axis_y, axis_z, angle };
108  }

493  template<typename T, int C = 4, int R = 4>
494  inline std::enable_if_t<(C >= 3 && R >= 3),
495  mat<decltype(tue::math::sin(std::declval<T>())), C, R>>
496  rotation_mat(
497  const T& axis_x, const T& axis_y, const T& axis_z,
498  const T& angle) noexcept
499  {
500  using U = decltype(tue::math::sin(angle));
501  decltype(auto) x = tue::nocopy_cast<U>(axis_x);
502  decltype(auto) y = tue::nocopy_cast<U>(axis_y);
503  decltype(auto) z = tue::nocopy_cast<U>(axis_z);
504 
505  U s, c;
506  tue::math::sincos(angle, s, c);
507  const auto omc = U(1) - c;
508 
509  const auto xx = x * x;
510  const auto xy = x * y;
511  const auto xz = x * z;
512  const auto xs = x * s;
513  const auto yy = y * y;
514  const auto yz = y * z;
515  const auto ys = y * s;
516  const auto zz = z * z;
517  const auto zs = z * s;
518  const auto xxomc = xx * omc;
519  const auto xyomc = xy * omc;
520  const auto xzomc = xz * omc;
521  const auto yyomc = yy * omc;
522  const auto yzomc = yz * omc;
523  const auto zzomc = zz * omc;
524 
525  return tue::detail_::mat_utils<U, C, R>::create(
526  xxomc + c, xyomc - zs, xzomc + ys, 0,
527  xyomc + zs, yyomc + c, yzomc - xs, 0,
528  xzomc - ys, yzomc + xs, zzomc + c, 0,
529  0, 0, 0, 1);
530  }
